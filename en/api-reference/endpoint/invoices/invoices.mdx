<ParamField
  header="Authorization"
  type="string"
  placeholder="Bearer {token}"
  required
>
  The Bearer authentication token is required to access this endpoint.
</ParamField>

<ParamField body="amount" type="number" placeholder="1.1" required>
  Sale amount to be created.
</ParamField>

<ParamField
  body="external_reference"
  type="string"
  placeholder="Referência externa"
>
  Use it to store an external reference to your Gates2b application, such as an
  ID. Optional field.
</ParamField>
<ParamField body="description" type="string" placeholder="Descrição da venda">
  Sale description. Optional field.
</ParamField>

<ParamField body="name_client" type="string" placeholder="Nome do pagador">
  Payer name. Optional field.
</ParamField>

<ParamField
  body="document_client"
  type="string"
  placeholder="CPF ou CNPJ do pagador"
>
  Payer document (CPF or CNPJ). Optional field.
</ParamField>
