<ParamField
  header="Authorization"
  type="string"
  placeholder="Bearer {token}"
  required
>
  The Bearer authentication token is required to access this endpoint.
</ParamField>

<ParamField
  body="clientMode"
  type="string"
  placeholder="noNeedToSpecify | fillDataNow | requestCheckout"
  required
>
  Defines how customer data is handled. - `noNeedToSpecify`: No customer data
  required. - `fillDataNow`: Customer data must be provided now. -
  `requestCheckout`: Customer data will be collected via checkout.
</ParamField>

<ParamField body="amount" type="number" placeholder="150.50" required>
  Payment amount.
</ParamField>

<ParamField
  body="expire_at"
  type="string"
  placeholder="2025-12-31T23:59:59Z"
  required
>
  Expiration date and time in ISO 8601 format.
</ParamField>

<ParamField
  body="description"
  type="string"
  placeholder="Descrição do pagamento"
>
  Payment description. Optional field.
</ParamField>

<ParamField
  body="external_reference"
  type="string"
  placeholder="Referência externa"
>
  External reference for your application. Optional field.
</ParamField>

<ParamField
  body="redirect_url"
  type="string"
  placeholder="https://sua-url.com/retorno"
>
  Redirect URL after payment. Optional field.
</ParamField>

<ParamField body="phoneNumber" type="string" placeholder="+5511999999999">
  Customer phone number. Optional field.
</ParamField>

<ParamField body="email" type="string" placeholder="cliente@email.com">
  Customer email. Optional field.
</ParamField>

<ParamField body="document_type" type="string" placeholder="CPF | CNPJ">
  Customer document type. Optional field.
</ParamField>

<ParamField body="name_client" type="string" placeholder="Nome do cliente">
  Customer name. Optional field.
</ParamField>

<ParamField body="document_client" type="string" placeholder="12345678900">
  Customer document. Optional field.
</ParamField>
